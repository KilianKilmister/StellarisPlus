
special_project = {
	key = "SURVEY_PLANET" #in loc get planet name
	icon = "gfx/interface/icons/situation_log/situation_log_scanning_beam.dds"
	event_chain = select_planet_chain
	cost = 0
	days_to_research = 30	#change
	tech_department = society_technology
	timelimit = -1
	picture = GFX_evt_enigmatic_fortress #change
	event_scope = ship_event

	requirements = {
		shipclass_science_ship = 1
		leader = scientist
		#skill = 1
	}


	on_success = {
		from = {
			enable_special_project = {
				owner = event_target:selector
				name = CONFIRM_PLANET
				location = THIS
			}
			event_target:selector = {
				abort_special_project = {
					location = prev
					type = SURVEY_PLANET
				}
			}
		}
	}
}

	on_success = {
			enable_special_project = {
				owner = event_target:selector
				name = FATE_TO_BE_DECIDED_GIVE_HOME
				location = this
			}
		event_target:selector = {
			abort_special_project = {
				type = CONFIRM_PLANET
			}
			abort_special_project = {
				type = SURVEY_PLANET
			}
			end_event_chain = select_planet_chain
			remove_point_of_interest = FATE_TO_BE_DECIDED_SELECT_HOME_POI
		}
	}
}
