# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

defined_text = {
	name = GetOptionM
	text = {
		trigger = { is_moon = yes fromfrom.owner = { has_technology = tech_climate_restoration } }
		localization_key = OptionM_yes

	}
	text = {
		trigger = { is_moon = no fromfrom.owner = { has_technology = tech_climate_restoration } }
		localization_key = OptionM_no
	}
	text = {
		trigger = { always = yes }
		localization_key = ""
	}
}

defined_text = {
	name = GetOptionGA
	text = {
		trigger = {
			habitability = {
				who = event_target:PHM_uplifted_hive
				value > 0.1
			}
			fromfrom.owner = { has_technology = tech_glandular_acclimation }
		}
		localization_key = OptionGA_yes

	}
	text = {
		trigger = {
			habitability = {
				who = event_target:PHM_uplifted_hive
				value <= 0.1
			}
			fromfrom.owner = { has_technology = tech_glandular_acclimation }
		}
		localization_key = OptionGA_no
	}
	text = {
		trigger = { always = yes }
		localization_key = ""
	}
}
